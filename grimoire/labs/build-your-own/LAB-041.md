## 📖 Background

Encrypted communication systems combine several cryptographic primitives into a secure channel. Understanding how these pieces fit together is essential for evaluating the security of any communication tool — from TLS to Signal to custom protocols. By building each layer yourself, you will understand why certain design choices matter and what can go wrong.

Key concepts:
- **Key exchange** — Two parties establish a shared secret over an untrusted channel (Diffie-Hellman / ECDH)
- **Symmetric encryption** — Fast encryption using the shared secret (AES-GCM provides both confidentiality and integrity)
- **Key derivation** — Turning a shared secret into encryption and MAC keys (HKDF)
- **Authentication** — Verifying message origin using HMAC or digital signatures
- **Message framing** — Structuring encrypted messages for reliable transmission over TCP

## 🔬 Lab Instructions

### Phase 1: Key Exchange and Key Derivation

1. Implement an ECDH key exchange module:
   - Generate ephemeral key pairs using the SECP256R1 curve
   - Exchange public keys between client and server
   - Derive the shared secret from the exchanged keys
2. Implement HKDF (HMAC-based Key Derivation Function):
   - Derive separate encryption and authentication keys from the shared secret
   - Use a unique salt and context info for domain separation
3. Test that both sides independently derive the same encryption keys
4. Display key fingerprints (first 8 bytes of SHA-256 of public key) for verification

### Phase 2: Encryption and Message Framing

1. Implement AES-256-GCM encryption:
   - Encrypt messages using the derived encryption key
   - Generate a unique nonce (96 bits) for every message
   - Include associated data (message sequence number) in the GCM authentication
2. Implement message framing for TCP transmission:
   - Frame format: `[4-byte length][12-byte nonce][ciphertext][16-byte GCM tag]`
   - Implement reliable send/receive that handles TCP stream fragmentation
3. Add a message sequence counter to detect replay attacks
4. Test encryption round-trip: plaintext → encrypt → transmit → decrypt → verify

### Phase 3: Secure Messaging Application

1. Build a TCP server that:
   - Listens for incoming connections
   - Performs the ECDH handshake with each client
   - Receives encrypted messages, decrypts, and displays them
   - Sends encrypted responses back to the client
2. Build a TCP client that:
   - Connects to the server and performs the ECDH handshake
   - Accepts user input, encrypts it, and sends it to the server
   - Receives and decrypts server responses
3. Implement session teardown with a secure close message
4. Test with two terminals: run the server in one, the client in the other

### Starter Framework
```python
#!/usr/bin/env python3
"""GRIMOIRE Secure Communication System — LAB-041 Build Your Own"""

import os
import struct
import socket
import hashlib
import threading

from cryptography.hazmat.primitives.asymmetric import ec
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.kdf.hkdf import HKDF
from cryptography.hazmat.primitives.ciphers.aead import AESGCM

# --- Key Exchange ---
class KeyExchange:
    """ECDH key exchange using SECP256R1."""

    def __init__(self):
        self.private_key = ec.generate_private_key(ec.SECP256R1())
        self.public_key = self.private_key.public_key()

    def get_public_bytes(self):
        """Serialize public key for transmission."""
        return self.public_key.public_bytes(
            serialization.Encoding.X962,
            serialization.PublicFormat.UncompressedPoint,
        )

    def derive_shared_secret(self, peer_public_bytes):
        """Derive shared secret from peer's public key."""
        peer_key = ec.EllipticCurvePublicKey.from_encoded_point(
            ec.SECP256R1(), peer_public_bytes
        )
        shared = self.private_key.exchange(ec.ECDH(), peer_key)
        return shared

    def fingerprint(self):
        """Return a short fingerprint of the public key for verification."""
        digest = hashlib.sha256(self.get_public_bytes()).hexdigest()
        return digest[:16]

# --- Key Derivation ---
def derive_keys(shared_secret, salt=None):
    """Derive encryption and authentication keys from shared secret using HKDF."""
    if salt is None:
        salt = b"\x00" * 32
    key_material = HKDF(
        algorithm=hashes.SHA256(),
        length=64,  # 32 bytes encryption + 32 bytes authentication
        salt=salt,
        info=b"grimoire-secure-comms-v1",
    ).derive(shared_secret)
    enc_key = key_material[:32]
    auth_key = key_material[32:]
    return enc_key, auth_key

# --- Encrypted Messaging ---
class SecureChannel:
    """Encrypt and decrypt messages using AES-256-GCM."""

    def __init__(self, enc_key):
        self.aesgcm = AESGCM(enc_key)
        self.send_seq = 0
        self.recv_seq = 0

    def encrypt(self, plaintext):
        """Encrypt a message and return framed bytes."""
        nonce = os.urandom(12)
        aad = struct.pack("!Q", self.send_seq)  # sequence number as associated data
        ciphertext = self.aesgcm.encrypt(nonce, plaintext.encode(), aad)
        # Frame: [4-byte total length][8-byte seq][12-byte nonce][ciphertext+tag]
        payload = aad + nonce + ciphertext
        frame = struct.pack("!I", len(payload)) + payload
        self.send_seq += 1
        return frame

    def decrypt(self, frame):
        """Decrypt a framed message and return plaintext."""
        # TODO: Parse the frame, extract seq/nonce/ciphertext, verify and decrypt
        payload_len = struct.unpack("!I", frame[:4])[0]
        payload = frame[4:4 + payload_len]
        seq = struct.unpack("!Q", payload[:8])[0]
        nonce = payload[8:20]
        ciphertext = payload[20:]
        aad = struct.pack("!Q", seq)
        # Verify sequence number
        if seq != self.recv_seq:
            raise ValueError(f"Sequence mismatch: expected {self.recv_seq}, got {seq}")
        plaintext = self.aesgcm.decrypt(nonce, ciphertext, aad)
        self.recv_seq += 1
        return plaintext.decode()

# --- Network Helpers ---
def send_msg(sock, data):
    """Send length-prefixed data over a socket."""
    sock.sendall(struct.pack("!I", len(data)) + data)

def recv_msg(sock):
    """Receive length-prefixed data from a socket."""
    raw_len = _recv_exact(sock, 4)
    if not raw_len:
        return None
    msg_len = struct.unpack("!I", raw_len)[0]
    return _recv_exact(sock, msg_len)

def _recv_exact(sock, n):
    """Receive exactly n bytes from a socket."""
    data = b""
    while len(data) < n:
        chunk = sock.recv(n - len(data))
        if not chunk:
            return None
        data += chunk
    return data

# --- Handshake ---
def perform_handshake_server(conn):
    """Server-side ECDH handshake."""
    kx = KeyExchange()
    # Receive client public key
    client_pub = recv_msg(conn)
    # Send server public key
    send_msg(conn, kx.get_public_bytes())
    shared = kx.derive_shared_secret(client_pub)
    enc_key, auth_key = derive_keys(shared)
    print(f"[handshake] Server key fingerprint: {kx.fingerprint()}")
    return SecureChannel(enc_key)

def perform_handshake_client(sock):
    """Client-side ECDH handshake."""
    kx = KeyExchange()
    # Send client public key
    send_msg(sock, kx.get_public_bytes())
    # Receive server public key
    server_pub = recv_msg(sock)
    shared = kx.derive_shared_secret(server_pub)
    enc_key, auth_key = derive_keys(shared)
    print(f"[handshake] Client key fingerprint: {kx.fingerprint()}")
    return SecureChannel(enc_key)

# --- Server ---
def run_server(host="127.0.0.1", port=9999):
    """Start the secure messaging server."""
    srv = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    srv.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    srv.bind((host, port))
    srv.listen(1)
    print(f"[server] Listening on {host}:{port}")
    conn, addr = srv.accept()
    print(f"[server] Connection from {addr}")
    channel = perform_handshake_server(conn)
    print("[server] Secure channel established.")
    try:
        while True:
            frame = recv_msg(conn)
            if not frame:
                break
            # Reconstruct the full frame with length prefix for decrypt
            full_frame = struct.pack("!I", len(frame)) + frame
            msg = channel.decrypt(full_frame)
            print(f"[received] {msg}")
    except Exception as e:
        print(f"[server] Error: {e}")
    finally:
        conn.close()
        srv.close()

# --- Client ---
def run_client(host="127.0.0.1", port=9999):
    """Start the secure messaging client."""
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.connect((host, port))
    print(f"[client] Connected to {host}:{port}")
    channel = perform_handshake_client(sock)
    print("[client] Secure channel established. Type messages (Ctrl+C to quit):")
    try:
        while True:
            msg = input("> ")
            if not msg:
                continue
            frame = channel.encrypt(msg)
            sock.sendall(frame)
    except (KeyboardInterrupt, EOFError):
        print("\n[client] Disconnecting.")
    finally:
        sock.close()

# --- Entry Point ---
if __name__ == "__main__":
    import sys
    if len(sys.argv) > 1 and sys.argv[1] == "server":
        run_server()
    else:
        run_client()
```
