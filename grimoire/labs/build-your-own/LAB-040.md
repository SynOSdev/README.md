## 📖 Background

Incident response follows a structured process: preparation, identification, containment, eradication, recovery, and lessons learned. The identification and containment phases depend heavily on fast, accurate data collection. Off-the-shelf IR tools are valuable, but understanding how evidence collection works at the code level ensures you can adapt to any environment — including systems where your favorite tools are not installed.

Key concepts:
- **Volatile evidence** — Data that is lost when a system powers down (running processes, network connections, memory contents)
- **Non-volatile evidence** — Persistent data (files, logs, registry entries, disk contents)
- **Order of volatility** — Collect the most volatile evidence first
- **Chain of custody** — Documenting every step of evidence handling
- **Hashing** — Verifying evidence has not been altered since collection

1. Create a `SystemSnapshot` class that collects:
   - System information: hostname, OS, kernel version, uptime, current time
   - Running processes: PID, name, user, command line, CPU%, memory%

   - Computes SHA-256 hashes for every collected file
   - Generates a manifest (filename, path, hash, size, timestamps)
3. Implement the order-of-volatility collection sequence: processes → connections → users → files → logs

### Phase 2: Analysis Modules

1. Create a `ProcessAnalyzer` that flags suspicious processes:
   - Processes running from temporary directories
   - Processes with no associated binary on disk
   - Processes consuming anomalous CPU or memory
   - Processes with suspicious names (encoded strings, random characters)
2. Create a `NetworkAnalyzer` that flags suspicious connections:
   - Connections to known-bad ports (e.g., 4444, 1337, unusual high ports)
   - Established connections to external IPs from unexpected processes
   - Listening services that are not in an expected baseline
3. Test both analyzers on your local system and review the flagged items

   - Accepts events from multiple sources (process start times, file modification times, log entries, connection timestamps)
   - Normalizes all timestamps to UTC
   - Sorts events chronologically
   - Outputs a unified timeline in CSV and JSON formats
2. Create a `ReportCompiler` that:
   - Combines system snapshot, analysis findings, and timeline into a single IR report
   - Includes an executive summary with key findings
   - Includes full evidence details with hashes
   - Outputs as both text and HTML
3. Run a full collection, analysis, and report cycle on your lab system

#!/usr/bin/env python3
"""GRIMOIRE Incident Response Toolkit — LAB-040 Build Your Own"""

import os
import csv
import json
import hashlib
import shutil
import socket
from datetime import datetime, timezone

import psutil

# --- Configuration ---
OUTPUT_DIR = f"ir-collection-{datetime.now().strftime('%Y%m%d-%H%M%S')}"

def ensure_output_dir():
    os.makedirs(OUTPUT_DIR, exist_ok=True)
    os.makedirs(os.path.join(OUTPUT_DIR, "evidence"), exist_ok=True)
    return OUTPUT_DIR

# --- Evidence Hashing ---
def hash_file(path, algorithm="sha256"):
    """Compute a cryptographic hash of a file."""
    h = hashlib.new(algorithm)
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()

# --- System Snapshot ---
class SystemSnapshot:
    """Collect volatile system information."""

    def collect(self):
        """Gather system state and return a structured dict."""
        snapshot = {
            "collected_at": datetime.now(timezone.utc).isoformat(),
            "hostname": socket.gethostname(),
            "boot_time": datetime.fromtimestamp(
                psutil.boot_time(), tz=timezone.utc
            ).isoformat(),
            "processes": self._collect_processes(),
            "connections": self._collect_connections(),
            "users": self._collect_users(),
        }
        return snapshot

    def _collect_processes(self):
        """Enumerate all running processes."""
        procs = []
        for proc in psutil.process_iter(
            ["pid", "name", "username", "cmdline", "cpu_percent", "memory_percent",
             "create_time", "exe", "status"]
        ):
            try:
                info = proc.info
                info["create_time_utc"] = datetime.fromtimestamp(
                    info.get("create_time", 0), tz=timezone.utc
                ).isoformat()
                procs.append(info)
            except (psutil.NoSuchProcess, psutil.AccessDenied):
                continue
        return procs

    def _collect_connections(self):
        """Enumerate network connections."""
        conns = []
        for c in psutil.net_connections(kind="inet"):
            conns.append({
                "local_addr": f"{c.laddr.ip}:{c.laddr.port}" if c.laddr else "",
                "remote_addr": f"{c.raddr.ip}:{c.raddr.port}" if c.raddr else "",
                "status": c.status,
                "pid": c.pid,
            })
        return conns

    def _collect_users(self):
        """List logged-in users."""
        return [
            {"name": u.name, "terminal": u.terminal, "host": u.host,
             "started": datetime.fromtimestamp(u.started, tz=timezone.utc).isoformat()}
            for u in psutil.users()
        ]

# --- File Collector ---
class FileCollector:
    """Copy evidence files and generate integrity manifest."""

    def __init__(self, output_dir):
        self.evidence_dir = os.path.join(output_dir, "evidence")
        self.manifest = []

    def collect_file(self, source_path):
        """Copy a file to the evidence directory and hash it."""
        if not os.path.isfile(source_path):
            return
        dest = os.path.join(self.evidence_dir, os.path.basename(source_path))
        shutil.copy2(source_path, dest)
        self.manifest.append({
            "original_path": source_path,
            "collected_path": dest,
            "sha256": hash_file(dest),
            "size": os.path.getsize(dest),
            "collected_at": datetime.now(timezone.utc).isoformat(),
        })

    def save_manifest(self):
        path = os.path.join(self.evidence_dir, "manifest.json")
        with open(path, "w") as f:
            json.dump(self.manifest, f, indent=2)

# --- Process Analyzer ---
class ProcessAnalyzer:
    """Flag suspicious processes from a collected process list."""

    SUSPICIOUS_DIRS = ["/tmp", "/var/tmp", "/dev/shm"]
    SUSPICIOUS_PORTS = {4444, 1337, 5555, 9001, 31337}

    def analyze(self, processes):
        """Return a list of flagged processes with reasons."""
        flagged = []
        for proc in processes:
            reasons = []
            exe = proc.get("exe") or ""
            # TODO: Implement suspicious-process checks
            # Check for execution from temp directories
            if any(exe.startswith(d) for d in self.SUSPICIOUS_DIRS):
                reasons.append("Executing from temporary directory")
            # Check for missing binary
            if exe and not os.path.isfile(exe):
                reasons.append("Binary not found on disk")
            # Check high resource usage
            if (proc.get("cpu_percent") or 0) > 80:
                reasons.append("High CPU usage")
            if (proc.get("memory_percent") or 0) > 50:
                reasons.append("High memory usage")
            if reasons:
                flagged.append({"process": proc, "reasons": reasons})
        return flagged

# --- Network Analyzer ---
class NetworkAnalyzer:
    """Flag suspicious network connections."""

    SUSPICIOUS_PORTS = {4444, 1337, 5555, 9001, 31337, 12345}

    def analyze(self, connections):
        """Return flagged connections with reasons."""
        flagged = []
        for conn in connections:
            reasons = []
            remote = conn.get("remote_addr", "")
            # TODO: Implement suspicious-connection checks
            if remote:
                port = int(remote.split(":")[-1]) if ":" in remote else 0
                if port in self.SUSPICIOUS_PORTS:
                    reasons.append(f"Connection to suspicious port {port}")
            if reasons:
                flagged.append({"connection": conn, "reasons": reasons})
        return flagged

# --- Timeline Generator ---
class TimelineGenerator:
    """Build a unified chronological event timeline."""

    def __init__(self):
        self.events = []

    def add_event(self, timestamp, source, description):
        """Add an event to the timeline."""
        self.events.append({
            "timestamp": timestamp,
            "source": source,
            "description": description,
        })

    def build_from_snapshot(self, snapshot):
        """Extract timeline events from a system snapshot."""
        for proc in snapshot.get("processes", []):
            self.add_event(
                proc.get("create_time_utc", ""),
                "process",
                f"Process started: {proc.get('name')} (PID {proc.get('pid')})"
            )

    def export(self, output_path):
        """Sort events and write timeline to CSV."""
        self.events.sort(key=lambda e: e["timestamp"])
        with open(output_path, "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=["timestamp", "source", "description"])
            writer.writeheader()
            writer.writerows(self.events)

# --- Report Compiler ---
class ReportCompiler:
    """Assemble the final IR report."""

    def compile(self, output_dir, snapshot, flagged_procs, flagged_conns, timeline_path):
        """Generate a text-based IR report."""
        report_path = os.path.join(output_dir, "ir-report.txt")
        # TODO: Implement full report compilation
        # Include: executive summary, system info, flagged processes,
        # flagged connections, timeline reference, evidence manifest
        with open(report_path, "w") as f:
            f.write("=" * 60 + "\n")
            f.write("GRIMOIRE INCIDENT RESPONSE REPORT\n")
            f.write("=" * 60 + "\n")
            f.write(f"Generated: {datetime.now(timezone.utc).isoformat()}\n")
            f.write(f"Hostname: {snapshot.get('hostname')}\n\n")
            f.write(f"Flagged Processes: {len(flagged_procs)}\n")
            f.write(f"Flagged Connections: {len(flagged_conns)}\n")
            f.write(f"Timeline: {timeline_path}\n")
        return report_path

# --- Entry Point ---
if __name__ == "__main__":
    print("[ir-toolkit] GRIMOIRE Incident Response Toolkit")
    out = ensure_output_dir()

    print("[phase-1] Collecting system snapshot...")
    snap = SystemSnapshot()
    snapshot = snap.collect()
    with open(os.path.join(out, "snapshot.json"), "w") as f:
        json.dump(snapshot, f, indent=2, default=str)

    print("[phase-2] Analyzing processes and connections...")
    proc_analyzer = ProcessAnalyzer()
    net_analyzer = NetworkAnalyzer()
    flagged_procs = proc_analyzer.analyze(snapshot["processes"])
    flagged_conns = net_analyzer.analyze(snapshot["connections"])
    print(f"  Flagged {len(flagged_procs)} processes, {len(flagged_conns)} connections")

    print("[phase-3] Generating timeline...")
    timeline = TimelineGenerator()
    timeline.build_from_snapshot(snapshot)
    timeline_path = os.path.join(out, "timeline.csv")
    timeline.export(timeline_path)

    print("[phase-4] Compiling report...")
    compiler = ReportCompiler()
    report = compiler.compile(out, snapshot, flagged_procs, flagged_conns, timeline_path)
    print(f"[done] IR report saved to {report}")
