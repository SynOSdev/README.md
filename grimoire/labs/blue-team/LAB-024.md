## 📖 Background

Threat intelligence is evidence-based knowledge about existing or emerging threats. The intelligence lifecycle — direction, collection, processing, analysis, dissemination, feedback — mirrors scientific methodology. At the tactical level, IOCs let defenders detect known-bad infrastructure and artifacts. At the strategic level, understanding adversary behavior drives security architecture decisions.

Key concepts:
- **IOC (Indicator of Compromise)** — Observable artifact indicating malicious activity
- **TTP (Tactics, Techniques, Procedures)** — How adversaries operate, mapped to frameworks like MITRE ATT&CK
- **STIX** — Structured Threat Information Expression — a language for describing threat intelligence
- **TAXII** — Trusted Automated Exchange of Indicator Information — transport protocol for STIX
- **Enrichment** — Adding context to raw indicators (geolocation, WHOIS, reputation scores)
- **Intelligence-driven defense** — Using threat knowledge to prioritize and focus defensive efforts

### Phase 3: Intelligence-Driven Detection

1. Convert IOCs into firewall block rules:
   ```bash
   # Generate iptables rules from malicious IPs
   jq -r '.indicators[] | select(.type == "ipv4") | .value' ~/grimoire/threat-intel/iocs/ioc_database.json | \
       while read -r ip; do
           echo "iptables -A INPUT -s $ip -j DROP"
           echo "iptables -A OUTPUT -d $ip -j DROP"
       done
   ```
2. Convert IOCs into IDS detection rules:
   ```bash
   # Generate Suricata rules from IOC database
   jq -r '.indicators[] | select(.type == "domain") | .value' ~/grimoire/threat-intel/iocs/ioc_database.json | \
       while read -r domain; do
           echo "alert dns any any -> any any (msg:\"GRIMOIRE TI - Malicious Domain: $domain\"; dns.query; content:\"$domain\"; nocase; sid:8000001; rev:1;)"
       done
   ```
3. Build an IOC lookup function for log analysis:
   ```bash
   # Check if any IPs in a log file match known malicious indicators
   comm -12 \
       <(jq -r '.indicators[] | select(.type == "ipv4") | .value' ~/grimoire/threat-intel/iocs/ioc_database.json | sort) \
       <(grep -oP '\b(?:\d{1,3}\.){3}\d{1,3}\b' ~/grimoire/siem-lab/logs/firewall.log | sort -u)
   ```
4. Create a threat intelligence brief summarizing your findings:
   - Adversary profile and known infrastructure

### Starter Framework
```python
#!/usr/bin/env python3
"""GRIMOIRE IOC Manager - LAB-024 Build Your Own"""

import json
import os
import re
import hashlib
from datetime import datetime
from collections import defaultdict

IOC_DIR = os.path.expanduser("~/.grimoire/threat-intel")
IOC_DB = os.path.join(IOC_DIR, "ioc_store.json")

os.makedirs(IOC_DIR, exist_ok=True)

IOC_TYPES = {
    "ipv4": re.compile(r'^(?:\d{1,3}\.){3}\d{1,3}$'),
    "domain": re.compile(r'^[a-zA-Z0-9]([a-zA-Z0-9-]*[a-zA-Z0-9])?(\.[a-zA-Z]{2,})+$'),
    "sha256": re.compile(r'^[a-fA-F0-9]{64}$'),
    "md5": re.compile(r'^[a-fA-F0-9]{32}$'),
    "url": re.compile(r'^https?://'),
}

def load_db():
    """Load the IOC database from disk."""
    if os.path.exists(IOC_DB):
        with open(IOC_DB, "r") as f:
            return json.load(f)
    return {"indicators": [], "metadata": {"created": datetime.now().isoformat()}}

def save_db(db):
    """Save the IOC database to disk."""
    db["metadata"]["updated"] = datetime.now().isoformat()
    with open(IOC_DB, "w") as f:
        json.dump(db, f, indent=2)

def detect_type(value):
    """Auto-detect the IOC type from its value."""
    for ioc_type, pattern in IOC_TYPES.items():
        if pattern.match(value):
            return ioc_type
    return "unknown"

def add_ioc(value, source="manual", tags=None, confidence="medium"):
    """Add an indicator to the database."""
    db = load_db()
    ioc_type = detect_type(value)

    # Check for duplicates
    for existing in db["indicators"]:
        if existing["value"] == value:
            print(f"[-] Duplicate IOC: {value}")
            return

    ioc = {
        "id": hashlib.sha256(value.encode()).hexdigest()[:12],
        "type": ioc_type,
        "value": value,
        "source": source,
        "tags": tags or [],
        "confidence": confidence,
        "first_seen": datetime.now().isoformat(),
        "last_seen": datetime.now().isoformat(),
        "sightings": 1,
    }
    db["indicators"].append(ioc)
    save_db(db)
    print(f"[+] Added {ioc_type} IOC: {value}")

def search_iocs(query):
    """Search IOCs by value, type, or tag."""
    db = load_db()
    results = []
    for ioc in db["indicators"]:
        if (query in ioc["value"] or
            query == ioc["type"] or
            query in ioc.get("tags", [])):
            results.append(ioc)
    return results

def export_firewall_rules(output_file=None):
    """Export IP-type IOCs as iptables block rules."""
    db = load_db()
    rules = []
    for ioc in db["indicators"]:
        if ioc["type"] == "ipv4":
            rules.append(f"iptables -A INPUT -s {ioc['value']} -j DROP")
            rules.append(f"iptables -A OUTPUT -d {ioc['value']} -j DROP")
    if output_file:
        with open(output_file, "w") as f:
            f.write("\n".join(rules) + "\n")
        print(f"[+] Exported {len(rules)} rules to {output_file}")
    else:
        for rule in rules:
            print(rule)

def export_ids_rules(output_file=None):
    """Export IOCs as Suricata detection rules."""
    db = load_db()
    rules = []
    sid = 9100001
    for ioc in db["indicators"]:
        if ioc["type"] == "domain":
            rules.append(
                f'alert dns any any -> any any (msg:"TI - Malicious Domain: {ioc["value"]}"; '
                f'dns.query; content:"{ioc["value"]}"; nocase; sid:{sid}; rev:1;)'
            )
            sid += 1
        elif ioc["type"] == "ipv4":
            rules.append(
                f'alert ip {ioc["value"]} any -> $HOME_NET any (msg:"TI - Malicious IP: {ioc["value"]}"; '
                f'sid:{sid}; rev:1;)'
            )
            sid += 1
    # TODO: Add URL and hash-based rules
    if output_file:
        with open(output_file, "w") as f:
            f.write("\n".join(rules) + "\n")
        print(f"[+] Exported {len(rules)} IDS rules to {output_file}")
    else:
        for rule in rules:
            print(rule)

def export_stix_bundle():
    """Export IOC database as a STIX 2.1 bundle."""
    db = load_db()
    objects = []
    for ioc in db["indicators"]:
        stix_pattern = ""
        if ioc["type"] == "ipv4":
            stix_pattern = f"[ipv4-addr:value = '{ioc['value']}']"
        elif ioc["type"] == "domain":
            stix_pattern = f"[domain-name:value = '{ioc['value']}']"
        elif ioc["type"] in ("sha256", "md5"):
            stix_pattern = f"[file:hashes.'{ioc['type'].upper()}' = '{ioc['value']}']"
        elif ioc["type"] == "url":
            stix_pattern = f"[url:value = '{ioc['value']}']"

        if stix_pattern:
            objects.append({
                "type": "indicator",
                "id": f"indicator--{ioc['id']}",
                "created": ioc["first_seen"],
                "modified": ioc["last_seen"],
                "name": f"{ioc['type'].upper()} - {ioc['value'][:40]}",
                "pattern": stix_pattern,
                "pattern_type": "stix",
                "valid_from": ioc["first_seen"],
                "labels": ioc.get("tags", []),
            })

    bundle = {"type": "bundle", "id": "bundle--grimoire-export", "objects": objects}
    print(json.dumps(bundle, indent=2))

def show_stats():
    """Display IOC database statistics."""
    db = load_db()
    type_counts = defaultdict(int)
    tag_counts = defaultdict(int)
    for ioc in db["indicators"]:
        type_counts[ioc["type"]] += 1
        for tag in ioc.get("tags", []):
            tag_counts[tag] += 1
    print(f"=== IOC Database Stats ===")
    print(f"Total indicators: {len(db['indicators'])}")
    print(f"\nBy type:")
    for t, c in sorted(type_counts.items()):
        print(f"  {t}: {c}")
    print(f"\nBy tag:")
    for t, c in sorted(tag_counts.items(), key=lambda x: -x[1]):
        print(f"  {t}: {c}")

if __name__ == "__main__":
    import sys
    if len(sys.argv) < 2:
        print("Usage: ioc-manager {add|search|enrich|export-fw|export-ids|export-stix|stats|help}")
        sys.exit(1)

    cmd = sys.argv[1]
    if cmd == "add":
        add_ioc(
            value=sys.argv[2] if len(sys.argv) > 2 else "",
            source=sys.argv[3] if len(sys.argv) > 3 else "manual",
        )
    elif cmd == "search":
        results = search_iocs(sys.argv[2] if len(sys.argv) > 2 else "")
        for r in results:
            print(f"  [{r['type']}] {r['value']} (confidence: {r['confidence']}, tags: {r['tags']})")
    elif cmd == "export-fw":
        export_firewall_rules(sys.argv[2] if len(sys.argv) > 2 else None)
    elif cmd == "export-ids":
        export_ids_rules(sys.argv[2] if len(sys.argv) > 2 else None)
    elif cmd == "export-stix":
        export_stix_bundle()
    elif cmd == "stats":
        show_stats()
    elif cmd == "help":
        print("GRIMOIRE IOC Manager - threat intelligence and indicator management")
    else:
        print(f"Unknown command: {cmd}")
```
